// DEMO 

cd /var/www/drmongo-demo
forever stop drmongo-demo
rm -rf bundle-copy-old
mv bundle-copy bundle-copy-old
cp -r bundle bundle-copy
rm -rf bundle
tar xfz drmongo-demo.tar.gz -C /var/www/drmongo-demo
mkdir bundle/programs/server/node_modules
rm -rf bundle/programs/server/npm/npm-bcrypt
npm install bcrypt
pushd bundle/programs/server/node_modules
rm -rf fibers
npm install fibers
npm install underscore
npm install source-map-support
npm install semver
popd
